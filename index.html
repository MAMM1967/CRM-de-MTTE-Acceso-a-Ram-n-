  <script>

    const CLIENT_ID = '232128500768-tmbbeonls89c7e0hmopfu999lrphbeco.apps.googleusercontent.com';

    const SPREADSHEET_ID = '1nKeeV1mDqtaHKdWIljitlcmck4b11sm0-RlSdeQj0YI';

    const TAB_NAME = 'Acciones Comerciales';

    

    let tokenClient;

    let allContacts = [];


    function gapiLoaded() {

      gapi.load('client', async () => {

        await gapi.client.init({

          discoveryDocs: [

            'https://www.googleapis.com/discovery/v1/apis/people/v1/rest',

            'https://sheets.googleapis.com/$discovery/rest?version=v4'

          ],

        });

      });

    }


    function gisLoaded() {

      tokenClient = google.accounts.oauth2.initTokenClient({

        client_id: CLIENT_ID, 

        scope: 'https://www.googleapis.com/auth/contacts https://www.googleapis.com/auth/spreadsheets',

        callback: '', 

      });

    }


    window.onload = () => {

      gapiLoaded();

      gisLoaded();

    };


    function handleAuthClick() {

      tokenClient.callback = async (resp) => {

        if (resp.error) return;

        document.getElementById('login-screen').style.display = 'none';

        document.getElementById('header').style.display = 'block';

        document.getElementById('main-content').style.display = 'block';

        await cargarContactos();

      };

      tokenClient.requestAccessToken();

    }


    async function cargarContactos() {

      const response = await gapi.client.people.people.connections.list({

        resourceName: 'people/me',

        pageSize: 100,

        personFields: 'names,organizations,emailAddresses,phoneNumbers',

      });

      allContacts = response.result.connections || [];

      renderizarTabla(allContacts);

    }


    function renderizarTabla(contactos) {

      const wrapper = document.getElementById('table-wrapper');

      let html = `<table><thead><tr><th>Nombre</th><th>Empresa</th><th>Acción</th></tr></thead><tbody>`;

      contactos.forEach(c => {

        html += `<tr>

          <td>${c.names?.[0]?.displayName || 'Sin nombre'}</td>

          <td>${c.organizations?.[0]?.name || '---'}</td>

          <td><button class="btn btn-primary" style="padding:5px 10px; font-size:0.8rem">Tratar</button></td>

        </tr>`;

      });

      html += `</tbody></table>`;

      wrapper.innerHTML = html;

    }


    function filtrarContactos() {

      const q = document.getElementById('search').value.toLowerCase();

      const res = allContacts.filter(c => (c.names?.[0]?.displayName || '').toLowerCase().includes(q));

      renderizarTabla(res);

    }
