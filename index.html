  <script>

    // CONFIGURACIÓN

    const CLIENT_ID = ''; // El entorno proporcionará el cliente si está configurado, o usa el tuyo.

    const SCOPES = 'https://www.googleapis.com/auth/contacts https://www.googleapis.com/auth/spreadsheets';

    const SPREADSHEET_ID = '1nKeeV1mDqtaHKdWIljitlcmck4b11sm0-RlSdeQj0YI';

    const SHEET_NAME = 'Acciones Comerciales';


    let tokenClient;

    let gapiInited = false;

    let gisInited = false;

    let allContacts = [];

    let selectedContact = null;


    // Inicialización de APIs

    function gapiLoaded() {

      gapi.load('client', async () => {

        await gapi.client.init({

          discoveryDocs: [

            'https://www.googleapis.com/discovery/v1/apis/people/v1/rest',

            'https://sheets.googleapis.com/$discovery/rest?version=v4'

          ],

        });

        gapiInited = true;

        checkBeforeStart();

      });

    }


    function gisLoaded() {

      tokenClient = google.accounts.oauth2.initTokenClient({

        client_id: CLIENT_ID || 'TU_CLIENT_ID_AQUI', 

        scope: SCOPES,

        callback: '', // definido al hacer click

      });

      gisInited = true;

      checkBeforeStart();

    }


    function checkBeforeStart() {

      if (gapiInited && gisInited) {

        // Listo para autenticar

      }

    }


    // Carga las funciones de Google al cargar la ventana

    window.onload = () => {

      gapiLoaded();

      gisLoaded();

    };


    function handleAuthClick() {

      tokenClient.callback = async (resp) => {

        if (resp.error !== undefined) throw (resp);

        document.getElementById('login-screen').style.display = 'none';

        document.getElementById('header').style.display = 'block';

        document.getElementById('main-content').style.display = 'block';

        await cargarContactos();

      };


      if (gapi.client.getToken() === null) {

        tokenClient.requestAccessToken({prompt: 'consent'});

      } else {

        tokenClient.requestAccessToken({prompt: ''});

      }

    }


    async function cargarContactos() {

      try {

        const response = await gapi.client.people.people.connections.list({

          resourceName: 'people/me',

          pageSize: 1000,

          personFields: 'names,organizations,emailAddresses,phoneNumbers,biographies,userDefined,urls,metadata',

        });

        allContacts = response.result.connections || [];

        renderTabla(allContacts);

        document.getElementById('loading').style.display = 'none';

      } catch (err) {

        console.error("Error cargando contactos:", err);

      }

    }


    function renderTabla(contactos) {

      const container = document.getElementById('table-container');

      if (contactos.length === 0) {

        container.innerHTML = "<p style='color:#9ca3af'>No se encontraron contactos.</p>";

        return;

      }


      let html = `<table>

        <thead>

          <tr>

            <th>Nombre</th>

            <th>Empresa</th>

            <th>Email</th>

            <th>Acciones</th>

          </tr>

        </thead>

        <tbody>`;


      contactos.forEach(c => {

        const nombre = c.names?.[0]?.displayName || 'Sin nombre';

        const empresa = c.organizations?.[0]?.name || '-';

        const email = c.emailAddresses?.[0]?.value || '-';

        

        html += `

          <tr>

            <td><strong>${nombre}</strong></td>

            <td>${empresa}</td>

            <td style="color:#9ca3af">${email}</td>

            <td>

              <button class="btn btn-primary" style="padding: 5px 12px; font-size: 0.75rem;" onclick="abrirAcciones('${c.resourceName}')">Tratar</button>

            </td>

          </tr>

        `;

      });


      html += `</tbody></table>`;

      container.innerHTML = html;

    }


    function filtrarContactos() {

      const query = document.getElementById('search').value.toLowerCase();

      const filtrados = allContacts.filter(c => {

        const nombre = (c.names?.[0]?.displayName || '').toLowerCase();

        const empresa = (c.organizations?.[0]?.name || '').toLowerCase();

        return nombre.includes(query) || empresa.includes(query);

      });

      renderTabla(filtrados);

    }


    function abrirAcciones(resourceName) {

      selectedContact = allContacts.find(c => c.resourceName === resourceName);

      document.getElementById('modal-title').textContent = `Acción: ${selectedContact.names?.[0]?.displayName}`;

      document.getElementById('action-modal').style.display = 'flex';

      document.getElementById('sync-status').style.display = 'none';

    }


    function cerrarModal() {

      document.getElementById('action-modal').style.display = 'none';

    }


    // LOGICA PRINCIPAL: Actualizar Contacto + Registrar en Sheet

    async function ejecutarAccion(tipo) {

      const statusDiv = document.getElementById('sync-status');

      statusDiv.style.display = 'block';

      statusDiv.className = 'status-msg';

      statusDiv.textContent = `Registrando ${tipo}...`;


      const ahora = new Date();

      const fechaStr = ahora.toLocaleString('es-ES');

      const nuevaNota = `\n[${fechaStr}] Acción realizada: ${tipo}.`;

      

      try {

        // 1. Actualizar en Google Contacts (API People)

        const currentBio = selectedContact.biographies?.[0]?.value || "";

        const updatedBio = currentBio + nuevaNota;


        await gapi.client.people.people.updateContact({

          resourceName: selectedContact.resourceName,

          updatePersonFields: 'biographies',

          resource: {

            etag: selectedContact.etag,

            biographies: [{ value: updatedBio, contentType: 'TEXT_PLAIN' }]

          }

        });


        // 2. Registrar en Google Sheets

        await registrarEnGoogleSheet(selectedContact, tipo, updatedBio);


        statusDiv.classList.add('success');

        statusDiv.textContent = "¡Registrado con éxito en Contactos y Excel!";

        

        setTimeout(() => {

          cerrarModal();

          cargarContactos(); // Recargar para tener el etag actualizado

        }, 1500);


      } catch (err) {

        console.error(err);

        statusDiv.style.color = "#ef4444";

        statusDiv.textContent = "Error al sincronizar. Revisa la consola.";

      }

    }


    async function registrarEnGoogleSheet(p, tipoAccion, notasCompletas) {

      // Mapeo de campos según tu solicitud

      const nombres = p.names?.[0] || {};

      const org = p.organizations?.[0] || {};

      const email = p.emailAddresses?.[0]?.value || "";

      

      // Separación de teléfonos

      const telfs = p.phoneNumbers || [];

      const tFijo = telfs.find(t => t.type === 'home' || t.type === 'work')?.value || "";

      const tMovil = telfs.find(t => t.type === 'mobile')?.value || telfs[0]?.value || "";


      // Dirección

      const addr = p.addresses?.[0] || {};

      

      // LinkedIn

      const linkedin = (p.urls || []).find(u => u.value.includes('linkedin.com'))?.value || "";


      // Etiquetas (User Defined)

      const etiquetas = (p.userDefined || []).map(u => u.value).join(', ');


      // Preparar la fila (Fecha, Etiquetas, Nombre, Segundo nombre, Apellidos, Empresa, Puesto, Correo, Tel Fijo, Tel Movil, País, Dirección, CP, Ciudad, Provincia, LinkedIn, Notas)

      const fila = [

        new Date().toLocaleDateString('es-ES'), // Fecha

        etiquetas,

        nombres.givenName || "",

        nombres.middleName || "",

        nombres.familyName || "",

        org.name || "",

        org.title || "",

        email,

        tFijo,

        tMovil,

        addr.country || "",

        addr.streetAddress || "",

        addr.postalCode || "",

        addr.city || "",

        addr.region || "",

        linkedin,

        `ACCIÓN: ${tipoAccion} | HISTORIAL: ${notasCompletas}`

      ];


      await gapi.client.sheets.spreadsheets.values.append({

        spreadsheetId: SPREADSHEET_ID,

        range: `${SHEET_NAME}!A:Q`,

        valueInputOption: 'USER_ENTERED',

        resource: {

          values: [fila]

        }

      });

    }
