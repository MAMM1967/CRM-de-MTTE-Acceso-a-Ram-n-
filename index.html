    <script>

        // Configuración y variables de estado

        let config = JSON.parse(localStorage.getItem('mtte_crm_v3')) || { token: '', googleId: '' };

        let currentTab = 'Agenda'; 

        let googleToken = null;

        let tokenClient = null;

        let session = { accessToken: null, dtableUuid: null, serverUrl: 'https://dtable-server.seatable.io' };


        // Nombres de tablas estándar

        let TABLE_NAMES = {

            CONTACTOS: 'Contactos',

            EMPRESAS: 'Empresas'

        };


        function updateStatus(msg, colorClass, dotColor) {

            const el = document.getElementById('connection-status');

            const dot = document.getElementById('status-dot');

            if (el) { el.innerText = msg; el.className = `text-[10px] ${colorClass} uppercase tracking-widest font-bold`; }

            if (dot) { dot.className = `w-2 h-2 rounded-full ${dotColor}`; }

        }


        async function authenticateSeaTable() {

            if (!config.token) return false;

            try {

                updateStatus('Conectando...', 'text-blue-400', 'bg-blue-400 animate-pulse');

                const res = await fetch('https://cloud.seatable.io/api/v2.1/dtable/app-access-token/', {

                    headers: { 'Authorization': `Token ${config.token.trim()}` }

                });

                if (!res.ok) throw new Error('Auth fail');

                const data = await res.json();

                session.accessToken = data.access_token;

                session.dtableUuid = data.dtable_uuid;

                session.serverUrl = data.dtable_server.replace(/\/$/, ""); 

                updateStatus('Conectado', 'text-emerald-500', 'bg-emerald-500');

                return true;

            } catch (e) {

                updateStatus('Error Token', 'text-red-500', 'bg-red-500');

                return false;

            }

        }


        // Función para obtener la lista real de tablas si hay error 404

        async function getAvailableTables() {

            try {

                const res = await fetch(`${session.serverUrl}/api/v1/dtables/${session.dtableUuid}/metadata/`, {

                    headers: { 'Authorization': `Token ${session.accessToken}` }

                });

                const data = await res.json();

                return data.metadata.tables.map(t => t.name);

            } catch (e) { return []; }

        }


        async function stRequest(tableName, method = 'GET', body = null) {

            if (!session.accessToken) {

                const ok = await authenticateSeaTable();

                if (!ok) return { error: 401 };

            }

            const url = `${session.serverUrl}/api/v1/dtables/${session.dtableUuid}/rows/?table_name=${encodeURIComponent(tableName)}`;

            try {

                const res = await fetch(url, {

                    method,

                    headers: { 

                        'Authorization': `Token ${session.accessToken}`, 

                        'Content-Type': 'application/json'

                    },

                    body: body ? JSON.stringify(body) : null

                });

                if (res.status === 404) return { error: 404, tableName };

                return await res.json();

            } catch (e) { return { error: 500 }; }

        }


        async function loadData() {

            const container = document.getElementById('st-content');

            container.innerHTML = '<div class="flex justify-center py-24"><div class="loading"></div></div>';

            

            const tableToFetch = (currentTab === 'Empresas') ? TABLE_NAMES.EMPRESAS : TABLE_NAMES.CONTACTOS;

            const data = await stRequest(tableToFetch);

            

            if (data.error === 404) {

                const available = await getAvailableTables();

                container.innerHTML = `

                    <div class="glass-card p-10 text-center border-red-500/30">

                        <h4 class="text-red-400 font-bold mb-4">Error 404: Tabla "${tableToFetch}" no encontrada</h4>

                        <p class="text-xs text-slate-400 mb-6">SeaTable dice que esa tabla no existe en esta Base de Datos.</p>

                        <div class="bg-black/40 p-4 rounded-xl inline-block text-left">

                            <p class="text-[10px] text-blue-400 font-bold uppercase mb-2">Tablas detectadas en tu SeaTable:</p>

                            <ul class="text-xs space-y-1">

                                ${available.map(t => `<li class="text-white">🔹 "${t}"</li>`).join('')}

                            </ul>

                        </div>

                        <p class="text-[10px] mt-6 text-slate-500 italic">Asegúrate de que el nombre coincida exactamente (mayúsculas, minúsculas y espacios).</p>

                    </div>`;

                return;

            }


            if (data.rows) renderData(data.rows);

        }


        function renderData(rows) {

            const container = document.getElementById('st-content');

            if (!rows.length) { 

                container.innerHTML = '<div class="text-center py-24 opacity-20"><p class="text-xs italic">Sin registros</p></div>'; 

                return; 

            }


            if (currentTab === 'Agenda') {

                container.innerHTML = rows.slice(0, 20).map(r => `

                    <div class="glass-card p-5 mb-4 flex justify-between items-center border-l-4 border-blue-500 hover:bg-white/5 transition-all">

                        <div class="flex items-center gap-4">

                            <div class="w-10 h-10 rounded-full bg-blue-500/10 flex items-center justify-center text-blue-400 font-bold text-xs">

                                ${r.Nombre ? r.Nombre[0] : '?'}

                            </div>

                            <div>

                                <h3 class="font-bold text-sm">${r.Nombre || ''} ${r['Apellido 1'] || ''}</h3>

                                <p class="text-[10px] text-slate-500 uppercase tracking-widest mt-0.5">${r['Nombre Empresa'] || 'PARTICULAR'}</p>

                            </div>

                        </div>

                    </div>

                `).join('');

            } else {

                const cols = Object.keys(rows[0]).filter(k => !k.startsWith('_'));

                container.innerHTML = `

                    <div class="glass-card overflow-hidden">

                        <div class="overflow-x-auto custom-scrollbar">

                            <table class="w-full text-[10px] text-left">

                                <thead class="bg-slate-800 text-blue-400 uppercase font-black">

                                    <tr>${cols.map(c => `<th class="p-5 whitespace-nowrap">${c}</th>`).join('')}</tr>

                                </thead>

                                <tbody class="divide-y divide-white/5">

                                    ${rows.map(r => `<tr class="hover:bg-white/[0.02]">${cols.map(c => `<td class="p-5 text-slate-300 whitespace-nowrap">${r[c] || '-'}</td>`).join('')}</tr>`).join('')}

                                </tbody>

                            </table>

                        </div>

                    </div>`;

            }

        }


        function changeTab(t) { 

            currentTab = t; 

            ['Agenda','Empresas','Contactos'].forEach(id => {

                const btn = document.getElementById('tab-'+id);

                if(btn) btn.className = (id === t) ? 'pb-3 tab-active tracking-widest transition-all' : 'pb-3 text-slate-500 hover:text-white tracking-widest transition-all';

            });

            loadData(); 

        }


        function toggleModal(id, show) { document.getElementById(id).classList.toggle('hidden', !show); }

        

        function saveConfig() {

            config.token = document.getElementById('st-token').value.trim();

            config.googleId = document.getElementById('google-id').value.trim();

            localStorage.setItem('mtte_crm_v3', JSON.stringify(config));

            location.reload();

        }


        window.onload = () => {

            if (config.token) loadData(); else toggleModal('config-modal', true);

        };
