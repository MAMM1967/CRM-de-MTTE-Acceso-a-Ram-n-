  <script>

    const CLIENT_ID = '232128500768-tmbbeonls89c7e0hmopfu999lrphbeco.apps.googleusercontent.com';

    const SPREADSHEET_ID = '1nKeeV1mDqtaHKdWIljitlcmck4b11sm0-RlSdeQj0YI';

    const TAB_NAME = 'Acciones Comerciales';

    

    let tokenClient;

    let allContacts = [];

    let selectedContact = null;


    // Inicialización

    function gapiLoaded() {

      gapi.load('client', async () => {

        await gapi.client.init({

          discoveryDocs: [

            'https://www.googleapis.com/discovery/v1/apis/people/v1/rest',

            'https://sheets.googleapis.com/$discovery/rest?version=v4'

          ],

        });

      });

    }


    function gisLoaded() {

      tokenClient = google.accounts.oauth2.initTokenClient({

        client_id: CLIENT_ID, 

        scope: 'https://www.googleapis.com/auth/contacts https://www.googleapis.com/auth/spreadsheets',

        callback: '', 

      });

    }


    window.onload = () => {

      gapiLoaded();

      gisLoaded();

    };


    function handleAuthClick() {

      tokenClient.callback = async (resp) => {

        if (resp.error) return;

        document.getElementById('login-screen').style.display = 'none';

        document.getElementById('header').style.display = 'block';

        document.getElementById('main-content').style.display = 'block';

        await cargarContactos();

      };

      tokenClient.requestAccessToken({prompt: 'consent'});

    }


    async function cargarContactos() {

      try {

        const response = await gapi.client.people.people.connections.list({

          resourceName: 'people/me',

          pageSize: 500,

          personFields: 'names,organizations,emailAddresses,phoneNumbers,biographies,userDefined,etag',

        });

        allContacts = response.result.connections || [];

        renderizarTabla(allContacts);

      } catch (e) {

        console.error(e);

      }

    }


    function renderizarTabla(contactos) {

      const wrapper = document.getElementById('table-wrapper');

      let html = `<table><thead><tr><th>Contacto / Empresa</th><th>Datos</th><th style="text-align:right">Gestión</th></tr></thead><tbody>`;

      

      contactos.forEach(c => {

        const nombre = c.names?.[0]?.displayName || 'Sin nombre';

        const empresa = c.organizations?.[0]?.name || '---';

        const email = c.emailAddresses?.[0]?.value || '-';

        const telf = c.phoneNumbers?.[0]?.value || '';


        html += `<tr>

          <td><div style="font-weight:700">${nombre}</div><div style="font-size:0.8rem; color:var(--text-muted)">${empresa}</div></td>

          <td><div style="font-size:0.9rem">${email}</div><div style="font-size:0.8rem; color:var(--text-muted)">${telf}</div></td>

          <td style="text-align:right">

            <button class="btn btn-primary" style="padding:6px 12px; font-size:0.8rem" onclick="abrirModal('${c.resourceName}')">Gestionar</button>

          </td>

        </tr>`;

      });

      

      html += `</tbody></table>`;

      wrapper.innerHTML = html;

    }


    function filtrarContactos() {

      const q = document.getElementById('search').value.toLowerCase();

      const filtrados = allContacts.filter(c => {

        const n = (c.names?.[0]?.displayName || '').toLowerCase();

        const e = (c.organizations?.[0]?.name || '').toLowerCase();

        return n.includes(q) || e.includes(q);

      });

      renderizarTabla(filtrados);

    }


    function abrirModal(resourceName) {

      selectedContact = allContacts.find(c => c.resourceName === resourceName);

      document.getElementById('modal-title').textContent = selectedContact.names?.[0]?.displayName;

      document.getElementById('modal-subtitle').textContent = selectedContact.organizations?.[0]?.name || 'Sin empresa';

      document.getElementById('action-modal').style.display = 'flex';

      document.getElementById('feedback').style.display = 'none';

    }


    function cerrarModal() {

      document.getElementById('action-modal').style.display = 'none';

    }


    async function procesarAccion(tipo) {

      const fb = document.getElementById('feedback');

      fb.style.display = 'block';

      fb.className = 'status-msg';

      fb.textContent = `Guardando ${tipo}...`;


      const fecha = new Date().toLocaleString('es-ES');

      const nuevaNota = `\n[${fecha}] Acción: ${tipo}`;


      try {

        // 1. Actualizar Notas en Google Contacts

        const bioActual = selectedContact.biographies?.[0]?.value || "";

        const bioNueva = bioActual + nuevaNota;


        const resUpdate = await gapi.client.people.people.updateContact({

          resourceName: selectedContact.resourceName,

          updatePersonFields: 'biographies',

          resource: {

            etag: selectedContact.etag,

            biographies: [{ value: bioNueva, contentType: 'TEXT_PLAIN' }]

          }

        });


        // Actualizar datos locales

        selectedContact.etag = resUpdate.result.etag;

        selectedContact.biographies = [{ value: bioNueva }];


        // 2. Registrar en Google Sheets

        await registrarEnSheets(selectedContact, tipo, bioNueva);


        fb.className = 'status-msg success';

        fb.textContent = "¡Registrado correctamente!";

        setTimeout(cerrarModal, 1500);


      } catch (err) {

        fb.className = 'status-msg error';

        fb.textContent = "Error al sincronizar.";

      }

    }


    async function registrarEnSheets(p, tipo, historialCompleto) {

      const row = [

        new Date().toLocaleDateString('es-ES'),

        (p.userDefined || []).map(u => u.value).join(', '),

        p.names?.[0]?.givenName || "",

        "",

        p.names?.[0]?.familyName || "",

        p.organizations?.[0]?.name || "",

        p.organizations?.[0]?.title || "",

        p.emailAddresses?.[0]?.value || "",

        "",

        p.phoneNumbers?.[0]?.value || "",

        "", "", "", "", "", "",

        `ACCIÓN: ${tipo} | NOTAS: ${historialCompleto}`

      ];


      await gapi.client.sheets.spreadsheets.values.append({

        spreadsheetId: SPREADSHEET_ID,

        range: `${TAB_NAME}!A:Q`,

        valueInputOption: 'USER_ENTERED',

        resource: { values: [row] }

      });

    }
