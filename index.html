  <!-- Firebase & Logic -->

  <script type="module">

    import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

    import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

    import { getFirestore, doc, setDoc, getDoc, collection, query, onSnapshot, updateDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


    // Configuración Base (Firebase se inicializará con la config del entorno)

    const firebaseConfig = JSON.parse(__firebase_config);

    const app = initializeApp(firebaseConfig);

    const auth = getAuth(app);

    const db = getFirestore(app);

    const appId = localStorage.getItem('crm_app_id') || (typeof __app_id !== 'undefined' ? __app_id : 'shared-crm-default');


    let user = null;

    let cloudContacts = [];

    let gTokenClient;


    // --- CONFIGURACIÓN DE UI ---

    window.saveSettings = () => {

      const gId = document.getElementById('cfg-client-id').value.trim();

      const aId = document.getElementById('cfg-app-id').value.trim();

      if (gId && aId) {

        localStorage.setItem('crm_client_id', gId);

        localStorage.setItem('crm_app_id', aId);

        location.reload();

      }

    };


    window.toggleSettings = () => {

      document.getElementById('setup-screen').classList.toggle('hidden');

    };


    // --- FIREBASE AUTH ---

    const initAuth = async () => {

      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

        await signInWithCustomToken(auth, __initial_auth_token);

      } else {

        await signInAnonymously(auth);

      }

    };


    onAuthStateChanged(auth, (u) => {

      user = u;

      if (user) {

        document.getElementById('main-ui').classList.remove('opacity-0');

        document.getElementById('user-status').textContent = `Conectado Cloud como: ${user.uid.substring(0,8)}...`;

        startRealtimeListener();

        initGoogleSDK();

      }

    });


    initAuth();


    // --- ESCUCHA EN TIEMPO REAL (Firestore) ---

    function startRealtimeListener() {

      if (!user) return;

      const q = collection(db, 'artifacts', appId, 'public', 'data', 'contacts');

      

      onSnapshot(q, (snapshot) => {

        cloudContacts = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

        renderTable();

      }, (error) => {

        console.error("Error Firestore:", error);

      });

    }


    // --- INTEGRACIÓN GOOGLE CONTACTS ---

    function initGoogleSDK() {

      const clientId = localStorage.getItem('crm_client_id');

      if (!clientId) {

        document.getElementById('setup-screen').classList.remove('hidden');

        return;

      }


      gTokenClient = google.accounts.oauth2.initTokenClient({

        client_id: clientId,

        scope: 'https://www.googleapis.com/auth/contacts.readonly',

        callback: async (resp) => {

          if (resp.access_token) {

            await syncGoogleToCloud(resp.access_token);

          }

        },

      });

    }


    window.handleAuthClick = () => {

      gTokenClient.requestAccessToken({ prompt: 'consent' });

    };


    async function syncGoogleToCloud(accessToken) {

      document.getElementById('sync-icon').classList.add('animate-spin');

      try {

        const res = await fetch('https://people.googleapis.com/v1/people/me/connections?personFields=names,emailAddresses,phoneNumbers,organizations,notes,urls', {

          headers: { 'Authorization': `Bearer ${accessToken}` }

        });

        const data = await res.json();

        const connections = data.connections || [];


        for (const p of connections) {

          const contactData = parseContact(p);

          // Subir a la nube usando el email o id de google como llave única

          const contactRef = doc(db, 'artifacts', appId, 'public', 'data', 'contacts', contactData.uniqueId);

          

          // Solo actualizamos si el contacto es nuevo o ha cambiado (merge)

          await setDoc(contactRef, {

            ...contactData,

            updatedBy: user.uid,

            lastSync: serverTimestamp()

          }, { merge: true });

        }

        

        alert(`Sincronización completada. ${connections.length} contactos procesados.`);

      } catch (e) {

        console.error(e);

        alert("Error en la sincronización.");

      } finally {

        document.getElementById('sync-icon').classList.remove('animate-spin');

      }

    }


    function parseContact(p) {

      const name = p.names?.[0]?.displayName || 'Sin nombre';

      const org = p.organizations?.[0] || {};

      const emails = p.emailAddresses || [];

      const phones = p.phoneNumbers || [];

      const email = emails[0]?.value || '';

      

      // Creamos un ID único basado en el nombre y empresa si no hay email

      const uniqueId = email ? email.replace(/[.#$[\]]/g, '_') : btoa(name + org.name).substring(0, 20);


      return {

        uniqueId,

        name,

        company: org.name || 'Individual',

        title: org.title || 'N/A',

        email: email,

        phone: phones[0]?.value || '',

        linkedin: (p.urls || []).find(u => u.value.includes('linkedin'))?.value || '',

        notes: p.notes?.[0]?.value || ''

      };

    }


    // --- RENDERIZADO DE TABLA ---

    window.renderTable = () => {

      const queryStr = document.getElementById('search-input').value.toLowerCase();

      const body = document.getElementById('contacts-body');

      

      const filtered = cloudContacts.filter(c => 

        c.name.toLowerCase().includes(queryStr) || 

        c.company.toLowerCase().includes(queryStr)

      );


      body.innerHTML = filtered.map(c => `

        <tr class="hover:bg-slate-800/30 transition-all animate-fade">

          <td class="p-4">

            <div class="font-bold text-white text-sm">${c.company}</div>

            <div class="text-slate-400 text-xs">${c.name} <span class="text-slate-600 ml-1">| ${c.title}</span></div>

          </td>

          <td class="p-4">

            <div class="flex gap-2">

              ${c.phone ? `<a href="https://wa.me/${c.phone.replace(/\D/g,'')}" target="_blank" class="text-emerald-500">💬</a>` : ''}

              ${c.email ? `<a href="mailto:${c.email}" class="text-blue-400">📧</a>` : ''}

              ${c.linkedin ? `<a href="${c.linkedin}" target="_blank" class="text-sky-500">🔗</a>` : ''}

            </div>

          </td>

          <td class="p-4">

            <div class="text-[10px] text-slate-500 max-w-xs truncate italic" title="${c.notes}">

              ${c.notes || '---'}

            </div>

          </td>

          <td class="p-4">

            <span class="text-[9px] bg-slate-700 px-2 py-1 rounded text-slate-300">

              ID: ${c.updatedBy ? c.updatedBy.substring(0,5) : 'Sistema'}

            </span>

          </td>

          <td class="p-4 text-right">

            <button onclick="editNotes('${c.uniqueId}', '${c.notes}')" class="text-[10px] border border-blue-500/30 text-blue-400 px-2 py-1 rounded hover:bg-blue-500 hover:text-white transition">NOTAS</button>

          </td>

        </tr>

      `).join('');


      // Actualizar contadores

      document.getElementById('stat-cloud').textContent = cloudContacts.length;

      document.getElementById('stat-mine').textContent = cloudContacts.filter(c => c.updatedBy === user?.uid).length;

    };


    window.editNotes = async (id, currentNotes) => {

      const newNotes = prompt("Añade una nota compartida para este contacto:", currentNotes);

      if (newNotes !== null) {

        const contactRef = doc(db, 'artifacts', appId, 'public', 'data', 'contacts', id);

        await updateDoc(contactRef, { 

          notes: newNotes,

          updatedBy: user.uid,

          lastSync: serverTimestamp() 

        });

      }

    };


  </script>
